/*Date function*/

SELECT GETDATE(); /*Returns the current date and time*/

SELECT SYSDATETIME();  /*Returns the current date and time with higher precision than GETDATE()*/

SELECT FORMAT(GETDATE(), 'yyyy-MM-dd HH:mm:ss'); /* Formats a date and time value*/

SELECT DATEPART(YEAR, GETDATE()); /*Returns a specified part of a date */

SELECT DATEADD(DAY, 7, GETDATE()); /*Adds or subtracts a specified time interval from a date*/

SELECT DATEDIFF(YEAR, '2001-11-29', GETDATE());  /*Returns the difference between two dates*/

SELECT CONVERT(VARCHAR, GETDATE(), 103); /*Converts a date and time value to a specified format*/

SELECT MONTH('2024-01-19'); /*Returns the month part of a date*/

SELECT DAY(GETDATE()); /*Returns the day of the month for a given date*/

SELECT YEAR('2024-01-19');  /*Returns the year part of a date*/
